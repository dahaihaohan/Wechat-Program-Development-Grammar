
一、视图容器：
		view:
			hover:
				{
					类型：Boolean
					默认值：false
					说明：是否启用单机态
				}
			hover-class:
				{
					类型：String
					默认值：none
					说明：指定按下去的样式类。当hover-class="none"时，没有单机态效果
				}
			hover-start-time:
				{
					类型：Number
					默认值：50
					说明：按住后多久出现单击态，单位毫秒
				}
			hover-stay-time:
				{
					类型：Number
					默认值：400
					说明：手指松开后单机态保留事件，单位毫秒
				}
		scroll-view可滚动视图：
			scroll-x:
				{
					类型：Boolean
					默认值：false
					说明：允许横向滚动
				}
			scroll-y:
				{
					类型：Boolean
					默认值：false
					说明：允许纵向滚动
				}
			upper-threshold:
				{
					类型：Number
					默认值：50
					说明：距顶部/左边多远时（单位px）触发scrolltoupper事件
				}
			lower-threshold:
				{
					类型：Number
					默认值：50
					说明：距底部/右边多远时（单位px）,触发scrolltolower事件
				}
			scroll-top:
				{
					类型：Number
					默认值：
					说明：设置竖向滚动条位置
				}
			scroll-left:
				{
					类型：Number
					默认值：
					说明：设置横向滚动条位置
				}
			scroll-into-view:
				{
					类型：String 
					默认值：
					说明：值应为某子元素id，则滚动到该元素，元素顶部对齐滚动区域顶部
				}
			bindscrolltouper:
				{
					类型：EventHandle
					默认值：
					说明：滚动到顶部/左边，会触发scrolltoupper事件
				}
			bindscrolltolower:
				{
					类型：EventHandle
					默认值：
					说明：滚动到底部/右边，会触发scrolltolower事件
				}
			bindscroll:
				{
					类型：EventHandle
					默认值：
					说明：滚动时触发，event.detail={scrollLeft,scrollTop,scrollHeight,scroll-Width,deltaX.deltaY}
				}
		swiper:
